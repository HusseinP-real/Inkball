package inkball;

public class alivePoint {
    public int x;
    public int y;

    public int getX() {
        return x;
    }

    public int getY() {
        return y;
    }

    public alivePoint(int x, int y) {
        this.x = x;
        this.y = y;
    }

    public void drawAlivePoint(App app) {
        String aPPath = "inkball/entrypoint.png";
        app.image(app.loadImage(aPPath), x, y);
    }


}
